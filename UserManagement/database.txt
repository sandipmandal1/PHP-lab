CREATE TABLE user_name (
    first_name VARCHAR(50),
    last_name VARCHAR(50),
    email VARCHAR(100),
    password VARCHAR(100),
    contact VARCHAR(20),
    gender VARCHAR(10),
    country VARCHAR(50),
    city VARCHAR(50),
    age INT,
    father_name VARCHAR(50),
    photo_data BLOB,
    user_type VARCHAR(50) DEFAULT 'student'
);
